package com.yourcompany.collegemanagement;

import com.yourcompany.collegemanagement.model.Student;
import com.yourcompany.collegemanagement.model.Course;
import com.yourcompany.collegemanagement.repository.StudentRepository;
import com.yourcompany.collegemanagement.repository.CourseRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.CommandLineRunner;
import org.springframework.stereotype.Component;

@Component
public class DataLoader implements CommandLineRunner {

    @Autowired
    private StudentRepository studentRepository;

    @Autowired
    private CourseRepository courseRepository;

    @Override
    public void run(String... args) throws Exception {
        // Add dummy students
        studentRepository.save(new Student("John", "Doe", "john.doe@example.com"));
        studentRepository.save(new Student("Jane", "Smith", "jane.smith@example.com"));

        // Add dummy courses
        courseRepository.save(new Course("Math 101", "MATH101"));
        courseRepository.save(new Course("History 201", "HIST201"));
    }
}
